\mintinline{bash}|ssh br-trainXX@isambard.gw4.ac.uk|
\item Password: \mintinline{bash}|openmpUG20|
\item From the bastion node, log in to Isambard Phase 1\newline
\mintinline{bash}|ssh phase1|

\mintinline{bash}|cd openmp-for-cs|
\end{enumerate}

\end{frame}

\begin{frame}
\frametitle{Using Isambard (2)}
\begin{enumerate}
\setcounter{enumi}{5}
\item Build the exercises\newline
\mintinline{bash}|make|
\item Submit a job\newline
\mintinline{bash}|qsub submit_stencil|
\item Check job status\newline
\mintinline{bash}|qstat -u $USER|
\item Check job output\newline
\mintinline{bash}|cat stencil.o9748|

\end{enumerate}
